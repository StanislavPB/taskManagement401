
Пользователь (User)
id (уникальный идентификатор пользователя, который присваивается в момент регистрации; int)
name (имя пользователя; String)
login (уникальный логин пользователя; String)
password (хешированный пароль пользователя; String)
status (статус пользователя: босс или сотрудник; int)
task (список задач, которые сотрудник выполняет; ArrayList)
sms (список сообщений, которыми обмениваются сотрудники; ArrayList)


Проект (Project)
id (уникальный идентификатор проекта; присваивается в момент создания; int)
name (название проекта; String)
description (описание проекта; String)
users (список сотрудников, работающих над проектом; ArrayList)
tasks (список задач, которые необходимо выполнить для завершения проекта; ArrayList)
status (статус выполнения проекта; boolean)


Задача (Task)
taskID (уникальный идентификатор задачи; присваивается в момент создания; Integer)
taskName (название задачи; String)
endDate (дата, к которой задача должна быть выполнена; LocalDate)
assignedUsers (список сотрудников, отвечающих за выполнение задачи; ArrayList)
priority (приоритет задачи: высокий, средний, низкий; Priority)
taskCompleted (статус выполнения задачи; boolean)
comments (список комментариев от сотрудников, которые работают над данной задачей; ArrayList)
project (проект, к которому относится данная задача; Project)


Комментарий (Comment)
id (уникальный идентификатор комментария; присваивается в момент создания; int)
user (автор комментария; User)
message (текст комментария; String)



Сообщение (Message)
id (уникальный идентификатор сообщения; присваивается в момент создания; int)
sender (ID отправителя; int)
receiver (ID получателя; int)
contents (текст сообщения; String)
sendTime (дата отправки сообщения; LocalDateTime)
